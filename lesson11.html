  case "admin":

  case "editor":
    console.log("You can edit content.");
    break;

    console.log("Read-only access.");

    console.log("Unknown role.");

üì¶ Real-World Use: Day of the Week

let day = new Date().getDay();

switch (day) {
  case 0:
    console.log("Sunday");
    break;
  case 1:
    console.log("Monday");
    break;
  case 6:
    console.log("Saturday");
    break;
  default:
    console.log("A working day");
}
üí¨ Interview Tip: Use switch when:
