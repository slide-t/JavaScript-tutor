}

üß† Challenge 2:
Write a function that capitalizes the first letter of any name input.
Example: "emeka" ‚û°Ô∏è "Emeka"
‚úÖ Show Solution 

function capitalizeName(name) {
  return name.charAt(0).toUpperCase() + name.slice(1);
}
console.log(capitalizeName("emeka")); // "Emeka"
