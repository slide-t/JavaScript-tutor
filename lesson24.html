üß† Challenge:
Create a function that checks if a user's name contains "admin" and formats the name properly.

‚úÖ Example: " admin_john " should become: "Admin John"
‚úÖ Show Solution 

function cleanName(input) {
  let trimmed = input.trim();
  if (trimmed.toLowerCase().includes("admin")) {
    return "Admin " + trimmed.replace(/admin_/i, "").replace("_", " ");
  }
  return trimmed;
}

console.log(cleanName("  admin_john ")); // Admin John
